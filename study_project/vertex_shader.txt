#version 330

layout(location = 0) in vec3 a_position;
layout(location = 1) in vec3 a_color;

uniform mat4 u_mvp;

out vec3 v_color;
out vec3 v_normal;

vec3 f(vec3 coord)
{
	coord.y = sin(coord.x)*cos(coord.z);
    return coord/8.0;
}

vec3 gradient(vec3 point)
{
  return vec3(-cos(point.x)*cos(point.z), 1, sin(point.x) * sin(point.z));
}

void main()
{
    v_color = a_color;
    v_normal = normalize(gradient(a_position));
    gl_Position = u_mvp * vec4(f(a_position),1.0);
}
